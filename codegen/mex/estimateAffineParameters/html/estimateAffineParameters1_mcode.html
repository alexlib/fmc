1function [AFFINEMATRIXELEMENTS, RESIDUALS, EXITFLAG] = estimateAffineParameters(IMAGE1, IMAGE2, SPATIALWINDOW, ROTATION, SCALING, TRANSLATIONY, TRANSLATIONX)
 2% estimateAffineParameters(IMAGE1, IMAGE2, ROTATION, SCALING, TRANSLATIONY,
 3% TRANSLATIONX) estimates the full affine matrix that maps IMAGE1 onto
 4% IMAGE2 using initial guesses for rotation, uniform scaling, and y- and x- translations.
 5
 6[height, width] = size(IMAGE1);
 7
 8% Window the second image
 9windowedImage2 = SPATIALWINDOW .* double(IMAGE2); 
10
11% Form an initial guess for the parameters of the full affine transformation matrix
12matrixElementsGuess = [SCALING, ROTATION, 0, 0, TRANSLATIONX, TRANSLATIONY];
13
14% Set lower bounds on the matrix elements
15lowerBounds = [0.4 -pi -0.5 -0.5 -width/2 -height/2];
16
17% Set upper bounds on the matrix elements
18upperBounds = [2.5 pi 0.5 0.5 width/2 height/2];
19
20opts.TolFun = 1E-6;
21opts.TolX = 1E-6;
22
23% Set solver options
24%opts = optimset('TolX', 1e-6, 'TolFun', 1e-6);
25
26% % Invoke the nonlinear least squares optimizaiton algorithm
27% [AFFINEMATRIXELEMENTS, RESIDUALS, EXITFLAG] = ...
28%     fmincon(@transformationResiduals, matrixElementsGuess,...
29%     [], [], [], [], lowerBounds, upperBounds, [],...
30%     opts, IMAGE1, windowedImage2, SPATIALWINDOW);
31
32% Invoke the nonlinear least squares optimizaiton algorithm
33[AFFINEMATRIXELEMENTS, RESIDUALS, EXITFLAG] = ...
34    lsqnonlin(@transformationResiduals, matrixElementsGuess,...
35    lowerBounds, upperBounds,...
36    opts, IMAGE1, windowedImage2, SPATIALWINDOW);
37
38
39end
