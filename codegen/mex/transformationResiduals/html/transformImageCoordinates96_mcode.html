1function [YOUT, XOUT] = transformImageCoordinates(TRANSFORM, XGRID, YGRID, METHOD, CENTER)
 2% TRANSFORM = affine matrix
 3% IMAGESIZE = size of image (pixels); IMAGESIZE = [nRows nCols]
 4
 5% Default to not cropping the image.
 6if nargin < 4
 7    METHOD = 'full';
 8end
 9
10% Count rows and columns in the original image 
11[nr, nc] = size(XGRID);
12
13% Determine the center of the transformation. Default to center of grid
14if nargin < 5
15    xc = (nc + 1) / 2;
16    yc = (nr + 1) / 2;
17else
18    yc = CENTER(1);
19    xc = CENTER(2);
20end
21
22% Shift grid origin to the center of grid
23xPoints = XGRID - xc;
24yPoints = YGRID - yc;
25
26% Reshape coordinate matrices into vectors
27xPointsVect = reshape(xPoints, 1, numel(xPoints));
28yPointsVect = reshape(yPoints, 1, numel(yPoints));
29
30% Apply the transformation matrix to the shifted points
31transformedPoints = TRANSFORM * [xPointsVect; yPointsVect; ones( 1, length(xPointsVect ) ) ];
32
33% Extract the X- and Y coordinates of the transformed points.
34Xt = transformedPoints(1, :);
35Yt = transformedPoints(2, :);
36
37% Turn transformed X- and Y- coordinates back into matrices and shift the origin back to (1, 1)
38x = reshape(Xt, nr, nc) + xc;
39y = reshape(Yt, nr, nc) + yc;
40
41% Crop the image if specified.
42if strcmp(METHOD, 'crop')
43    XOUT = (x(x >= min(XGRID(:)) & x <= max(XGRID(:)) & y >= min(YGRID(:)) & y <= max(YGRID(:))));
44    YOUT = (y(x >= min(XGRID(:)) & x <= max(XGRID(:)) & y >= min(YGRID(:)) & y <= max(YGRID(:))));
45    
46else % Don't crop the image if cropping wasn't specified.
47    XOUT = x;
48    YOUT = y;
49    
50end
51
52
53end
